X <- as.matrix(protein_mat)  
y <- metadata_df$LRTI_adjudication == "Definite"

# Function to extract non-zero values
lasso_coef_df <- function(mod) {
  coefs <- coef(mod, s = "lambda.1se")[, 1]
  
  # Remove intercept
  coefs <- coefs[-1]
  
  # Identify nonzero coefficients
  nonzero <- coefs[coefs != 0]
  
  # If fewer than 3 proteins selected, pick the top 3 by absolute value
  if(length(nonzero) < 3){
    abs_coefs <- abs(coefs)
    top3 <- sort(abs_coefs, decreasing = TRUE)[1:3]
    nonzero <- coefs[names(top3)]
  }
  
  # Return as data.frame
  data.frame(protein = names(nonzero), coef = as.numeric(nonzero))
}

# Function for a single outer CV fold
lasso_cv_outer_fold <- function(test_fold) {
  train <- metadata_df$fold != test_fold
  foldid <- metadata_df$fold[train]
  foldid <- as.integer(as.factor(foldid))

  mod <- glmnet::cv.glmnet(
    X[train, ], y[train],
    family = "binomial",
    foldid = foldid
  )

  # Predictions on test fold
  preds <- predict(mod, X[!train, ], type = "response", s = "lambda.1se")[, 1] %>%
    data.frame(pred = .) %>%
    tibble::rownames_to_column("sample_name")

  # Nonzero coefficients for this fold
  nonzero <- lasso_coef_df(mod) %>%
    mutate(fold = test_fold)

  list(pred = preds, mod = mod, coef = nonzero)
}

# Run LASSO across all folds
lasso_results <- lapply(1:max(metadata_df$fold), lasso_cv_outer_fold)

# Organize predictions and coefficients
lasso_preds <- lasso_results %>%
  lapply(function(x) x$pred) %>%
  bind_rows()

lasso_features <- lasso_results %>%
  lapply(function(x) x$coef) %>%
  bind_rows()

# Save outputs
write.csv(lasso_preds, file.path(data_dir, "lasso_preds.csv"), row.names = FALSE)
write.csv(lasso_features, file.path(data_dir, "lasso_fold_coefs.csv"), row.names = FALSE)

# Import data
metadata_df <- read.csv(file.path(data_dir, "fake_metadata_with_folds.csv"), stringsAsFactors = FALSE)
protein_mat  <- read.csv(file.path(data_dir, "fake_protein_matrix.csv"), row.names = 1)

# Import selected proteins per fold
lasso_features <- read.csv(file.path(data_dir, "lasso_fold_coefs.csv"), stringsAsFactors = FALSE)

# Ensure row names match sample names (safety step)
rownames(protein_mat) <- metadata_df$sample_name

# Outcome variable
y <- as.factor(metadata_df$LRTI_adjudication)  

# Function to run Random Forest for one outer CV fold
lassoRF_cv_outer_fold <- function(test_fold) {
  
  # Extract proteins selected by LASSO for this fold
  keep <- lasso_features %>%
            filter(fold == test_fold) %>%
            .$protein
  
  # Subset protein matrix to selected proteins; rows = samples, columns = selected proteins
  X <- protein_mat[, keep, drop = FALSE]  
  
  # Boolean for training samples
  train <- metadata_df$fold != test_fold
  
  # Fit Random Forest (10000 Trees)
  rf <- randomForest(X[train, , drop = FALSE], y[train], ntree = 10000)
  
  # Predictions on test fold
  preds <- predict(rf, newdata = X[!train, , drop = FALSE], type = "prob")[, "Definite"]
  
  list(test_fold = test_fold, mod = rf, pred = preds)
}

# Run RF across all folds
cv_list <- lapply(1:max(metadata_df$fold), lassoRF_cv_outer_fold)

# Collect out-of-fold predictions for all samples
rf_preds <- cv_list %>%
  lapply(function(x) {
    data.frame(
      sample_name = rownames(protein_mat)[metadata_df$fold == x$test_fold],
      pred = x$pred
    )
  }) %>%
  bind_rows() %>%
  left_join(metadata_df %>% select(sample_name, LRTI_adjudication), by = "sample_name") %>%
  relocate(LRTI_adjudication, .after = sample_name)

# Save RF predictions
write.csv(rf_preds, file.path(data_dir, "lassoRF_preds.csv"), row.names = FALSE)

# Collect RF votes (training set OOB probabilities)
rf_votes <- cv_list %>%
  lapply(function(x) {
    as.data.frame(x$mod$votes) %>%
      tibble::rownames_to_column("sample_name") %>%
      mutate(test_fold = x$test_fold) %>%
      rename(pred = Definite) %>%
      select(test_fold, sample_name, pred)
  }) %>%
  bind_rows() %>%
  left_join(metadata_df %>% select(sample_name, LRTI_adjudication), by = "sample_name") %>%
  relocate(LRTI_adjudication, .after = sample_name)

# Save votes
write.csv(rf_votes, file.path(data_dir, "lassoRF_votes.csv"), row.names = FALSE)
